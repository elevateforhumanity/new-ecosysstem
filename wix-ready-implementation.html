// Add this to your Wix Code (Corvid) $w.onReady(function () { function updateCountdown() { const now = new Date().getTime(); const cohortStart = new Date(); cohortStart.setDate(cohortStart.getDate() + 7); // 7 days from now cohortStart.setHours(0, 0, 0, 0); // Start of day const distance = cohortStart - now; if (distance > 0) { const days = Math.floor(distance / (1000 * 60 * 60 * 24)); const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60)); const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60)); const seconds = Math.floor((distance % (1000 * 60)) / 1000); $w("#countdown").text = `${days.toString().padStart(2, '0')}:${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`; } else { $w("#countdown").text = "ENROLLMENT OPEN!"; } } setInterval(updateCountdown, 1000); updateCountdown(); }); // Note: Make sure to give your countdown text element the ID "countdown" in Wix Editor
